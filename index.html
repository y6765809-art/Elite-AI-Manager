import express from "express"; import fetch from "node-fetch"; import multer from "multer"; import fs from "fs"; const app = express(); const upload = multer({ dest: "uploads/" }); const API_KEY = process.env.CLAUDE_API_KEY; app.use(express.json()); // ===== FRONTEND בתוך השרת ===== app.get("/", (req, res) => { res.send(`

Elite AI 🔥





  

  
  
  `);
});

// ===== CHAT =====
app.post("/chat", async (req, res) => {
  const { message, system } = req.body;

  const r = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "x-api-key": API_KEY,
      "anthropic-version": "2023-06-01",
      "content-type": "application/json"
    },
    body: JSON.stringify({
      model: "claude-3-5-sonnet-20240620",
      max_tokens: 2000,
      system,
      messages: [{ role: "user", content: message }]
    })
  });

  res.json(await r.json());
});

// ===== FILE =====
app.post("/upload", upload.single("file"), async (req, res) => {
  const content = fs.readFileSync(req.file.path, "utf-8");

  const r = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "x-api-key": API_KEY,
      "anthropic-version": "2023-06-01",
      "content-type": "application/json"
    },
    body: JSON.stringify({
      model: "claude-3-5-sonnet-20240620",
      max_tokens: 2000,
      messages: [{
        role: "user",
        content: \`נתח את הקובץ:\\n\${content}\`
      }]
    })
  });

  res.json(await r.json());
});

app.listen(3000, () => console.log("🔥 http://localhost:3000"));
